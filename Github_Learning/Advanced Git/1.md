# View Commit History

git log

# Revert Commit

git revert commit_id

# Reset Commit

Soft Reset:

git reset --soft HEAD~1

# Hard Reset:

git reset --hard HEAD~1

# Stash

Save unfinished work:

git stash

# View stash:

git stash list

# Restore:

git stash pop

# Tagging

Create tag:

git tag v1.0

Push tag:

git push origin v1.0
